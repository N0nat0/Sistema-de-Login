CREATE TABLE `bd_empresa`.`tb_cliente` ( `cod_cliente` INT(11) NOT NULL AUTO_INCREMENT , `nm_cliente` VARCHAR(255) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL , `login_cliente` VARCHAR(50) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL , `senha_client` VARCHAR(32) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL , PRIMARY KEY (`cod_cliente`)) ENGINE = InnoDB;
